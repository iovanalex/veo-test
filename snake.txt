/* Private macro */
#define B_WIDTH 8
#define B_HEIGHT 8
#define SPEED 50

/* Private variables */

enum direction{UP, RIGHT, DOWN, LEFT};

typedef struct Points{
	int x;
	int y;
} Points_t;

typedef struct Food{
	Points_t poz;
}Food_t;

/*typedef struct Blocks{
	Points_t center;
	int width;
	int height;
} Blocks_t;*/

typedef struct Snake{
	int speed;
	enum direction dir;
	Points_t tail[200];
	int lenght;
} Snake_t;

/* Private function prototypes */
int displayInit(void);
void gameSetup(void);
void drawSnake(void);
void snakeLogic(void);
void foodLogic(void);
void drawFood(void);
void runGame(void);

/* Private functions */



Snake_t snake1;
Food_t food;
static TP_STATE* TP_State;

**===========================================================================

int main(void)
{
  /**
  *  IMPORTANT NOTE!
  *  The symbol VECT_TAB_SRAM needs to be defined when building the project
  *  if code has been located to RAM and interrupts are used. 
  *  Otherwise the interrupt table located in flash will be used.

  *  SCB->VTOR register.  
  *  E.g.  SCB->VTOR = 0x20000000;  
  */

  /* TODO - Add your application code here */
	displayInit();
	gameSetup();
	unsigned long time=0;
  /* Game LOOP */
	while (1)
	{
		touchHandle();
		if(time%(SPEED)==0){
			LCD_Clear(LCD_COLOR_BLACK);
			runGame();
			drawSnake();
			drawFood();
		}
		time++;
	}
}

void touchHandle(){
	TP_State = IOE_TP_GetState();
	if (TP_State->TouchDetected) {
		int X;
		int Y;
		Y = TP_State->Y;
		X = TP_State->X;
		if(X>LCD_PIXEL_WIDTH/3&&X<LCD_PIXEL_WIDTH/3*2){
			if(Y<LCD_PIXEL_HEIGHT/3&&Y>0){
				snake1.dir=UP;
			}
			if(Y>LCD_PIXEL_HEIGHT/3*2&&Y<LCD_PIXEL_HEIGHT){
				snake1.dir=DOWN;
			}
		}
		if(Y<LCD_PIXEL_HEIGHT/3*2&&Y>LCD_PIXEL_HEIGHT/3){
			if(X<LCD_PIXEL_WIDTH/3&&X>0){
				snake1.dir=LEFT;
			}
			if(X>LCD_PIXEL_WIDTH/3*2&&X<LCD_PIXEL_WIDTH-1){
				snake1.dir=RIGHT;
			}
		}
	}
}

void runGame(){
	snakeLogic();
	if(snake1.tail[0].x==food.poz.x&&snake1.tail[0].y==food.poz.y){
		foodLogic();
		drawFood();
		snake1.lenght++;
	}
	if(snake1.tail[0].x>LCD_PIXEL_WIDTH-B_WIDTH||snake1.tail[0].y>LCD_PIXEL_HEIGHT-B_HEIGHT){
		LCD_DisplayStringLine(LINE(2),"You died");
		while(1){}
	}
	if(snake1.tail[0].x<0||snake1.tail[0].y<0){
		LCD_DisplayStringLine(LINE(2),"You died");
		while(1){}
	}

}

void foodLogic(){
	int x=rand()%(LCD_PIXEL_WIDTH/B_WIDTH);
	int y=rand()%(LCD_PIXEL_HEIGHT/B_HEIGHT);
	food.poz.x=x*B_WIDTH;
	food.poz.y=y*B_HEIGHT;
}

void snakeLogic(){
	for(int i=snake1.lenght; i>0;i--){
			snake1.tail[i]=snake1.tail[i-1];
	}
	switch (snake1.dir){
	case UP:
		snake1.tail[0].y=snake1.tail[0].y-B_HEIGHT;
		break;
	case RIGHT:
		snake1.tail[0].x=snake1.tail[0].x+B_WIDTH;
		break;
	case DOWN:
		snake1.tail[0].y=snake1.tail[0].y+B_HEIGHT;
		break;
	case LEFT:
		snake1.tail[0].x=snake1.tail[0].x-B_WIDTH;
		break;
	}


}

void drawFood(){
	LCD_DrawFullRect(food.poz.x,food.poz.y,B_WIDTH,B_HEIGHT);
}

void drawSnake(){
	for(int i=0; i<snake1.lenght; i++){
		LCD_DrawFullRect(snake1.tail[i].x, snake1.tail[i].y,B_WIDTH,B_HEIGHT);
	}
}

void gameSetup(){
	foodLogic();
	snake1.tail[0].x=LCD_PIXEL_WIDTH/30*2;
	snake1.tail[0].y=LCD_PIXEL_HEIGHT/40*2;
	snake1.tail[1].x=snake1.tail[0].x;
	snake1.tail[1].y=snake1.tail[0].y+B_HEIGHT;
	snake1.dir=RIGHT;
	snake1.lenght=2;
	for(int i=2;i<200;i++){
		snake1.tail[i].x=-1;
		snake1.tail[i].y=-1;
	}
}

void myDelay(int speed){
	for(int i=0; i<= speed*100000; i++){
		;
	}
}

int displayInit(void){

	LCD_Init();
	LCD_SetColors(LCD_COLOR_GREEN, LCD_COLOR_BLACK);
	LCD_LayerInit();
	LCD_SetLayer(LCD_FOREGROUND_LAYER);
	LTDC_Cmd(ENABLE);
	LCD_SetFont(&Font16x24);
	LCD_DisplayChar(LINE(5), 20, '*');
	LCD_Clear(LCD_COLOR_BLACK);
	if (IOE_Config() == IOE_OK) {
		return 1;
	}else{
		LCD_DisplayStringLine(LINE(1), "ERROR");
		while(1){}
	}
}
